# PWC estimator algorithm run on data generated from a distrution composed
# by bivariate uniform distributions (a discrete mixture of uniforms with equal weights):
# 1) uniform on [-1, 1]x[-1, 1]
# 2) uniform on [6, 8]x[10, 33]
# 3) uniform on [1, 2]x[1, 2]
# 4) uniform on [2, 2.5]x[10, 24]

np.random.seed(22)
u1 = np.random.uniform(-1, 1, size=(300, 2))
u2 = np.concatenate((np.random.uniform(6, 8, size=(300, 1)),
                    np.random.uniform(10, 33, size=(300, 1))), axis=1)
u3 = np.random.uniform(1, 2, size=(300, 2))
u4 = np.concatenate((np.random.uniform(2, 2.5, size=(300, 1)),
                    np.random.uniform(10, 24, size=(300, 1))), axis=1)
xs = np.concatenate((u1, u2, u3, u4), axis=0)
In [3]:
# create PWC distribution class object and run method to build PWC estimator
pwc1 = pwc.PWCDistribution(xs)
pwc1.PWCEstimator(verbose=False)
pwc1.plot(showdata=True);
